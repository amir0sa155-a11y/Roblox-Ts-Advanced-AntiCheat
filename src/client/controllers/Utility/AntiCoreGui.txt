import { CoreController } from "../Main/Core";
import { Controller, OnStart } from "@flamework/core";
import { Players, HttpService, UserInputService, GuiService, Workspace, RunService } from "@rbxts/services";
import { CoreGuiController } from "./CoreGuiHide";

@Controller()
export class GuiAntiCheatController implements OnStart {
	constructor(private security: CoreController) {}

	public async onStart() {
		while (!CoreGuiController.Ready) {
			task.wait();
		}

		const localPlayer = Players.LocalPlayer;
		if (!localPlayer) return;

		const playerGui = localPlayer.WaitForChild("PlayerGui") as PlayerGui;
		const camera = Workspace.CurrentCamera || (Workspace.WaitForChild("Camera") as Camera);

		const CheckUi = (ui: Instance, expectedParent: Instance, isScreenGui: boolean) => {
			ui.AncestryChanged.Connect((_, newParent) => {
				if (newParent !== expectedParent) {
					this.security.fireBan("lb1");
				}
			});

			task.spawn(() => {
				while (ui.Parent === expectedParent) {
					task.wait(0.6);
					ui.Name = HttpService.GenerateGUID(false);
					if (isScreenGui) {
						(ui as ScreenGui).DisplayOrder = math.random(85, 115);
					} else {
						(ui as GuiObject).ZIndex = math.random(85, 115);
					}
				}
			});
		};

		const size = 44;
		const targetSize = 36.5;
		const Height = 52;
		const minWidth = 160;
		const maxShownDropDowns = 6;
		const margin = 12;
		const dropdownPositionX = 16;
		const dropdownPositionY = size - 38;

		const screenGui = new Instance("ScreenGui");
		screenGui.Name = HttpService.GenerateGUID(false);
		screenGui.ResetOnSpawn = false;
		screenGui.IgnoreGuiInset = false;
		screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling;
		screenGui.Parent = playerGui;
		CheckUi(screenGui, playerGui, true);

		const container = new Instance("Frame");
		container.Name = HttpService.GenerateGUID(false);
		container.AnchorPoint = new Vector2(0.5, 0.5);
		container.Position = UDim2.fromOffset(76.6 + targetSize / 2, -42.3 + targetSize / 2);
		container.Size = UDim2.fromOffset(targetSize, targetSize);
		container.BackgroundTransparency = 1;
		container.Parent = screenGui;
		CheckUi(container, screenGui, false);

		const ratio = new Instance("UIAspectRatioConstraint");
		ratio.AspectRatio = 1;
		ratio.AspectType = Enum.AspectType.ScaleWithParentSize;
		ratio.Parent = container;

		const button = new Instance("TextButton");
		button.Name = HttpService.GenerateGUID(false);
		button.Size = UDim2.fromScale(1, 1);
		button.BackgroundTransparency = 1;
		button.TextTransparency = 1;
		button.Parent = container;
		CheckUi(button, container, false);

		const dropdown = new Instance("Frame");
		dropdown.Name = HttpService.GenerateGUID(false);
		dropdown.AutomaticSize = Enum.AutomaticSize.X;
		dropdown.AnchorPoint = new Vector2(0, 0);
		dropdown.Position = UDim2.fromOffset(dropdownPositionX, dropdownPositionY);
		dropdown.Size = UDim2.fromOffset(0, 0);
		dropdown.ClipsDescendants = true;
		dropdown.BackgroundTransparency = 1;
		dropdown.Visible = false;
		dropdown.Parent = screenGui;
		CheckUi(dropdown, screenGui, false);

		const Corner = new Instance("UICorner");
		Corner.CornerRadius = new UDim(50, 50);
		Corner.Parent = button;

		const Corner2 = new Instance("UICorner");
		Corner2.CornerRadius = new UDim(50, 50);
		Corner2.Parent = container;

		const scroller = new Instance("ScrollingFrame");
		scroller.Name = HttpService.GenerateGUID(false);
		scroller.AutomaticSize = Enum.AutomaticSize.X;
		scroller.AutomaticCanvasSize = Enum.AutomaticSize.Y;
		scroller.Position = UDim2.fromOffset(0, 0);
		scroller.Size = UDim2.fromOffset(0, 0);
		scroller.CanvasSize = UDim2.fromOffset(0, 0);
		scroller.ScrollingEnabled = true;
		scroller.VerticalScrollBarInset = Enum.ScrollBarInset.None;
		scroller.BackgroundTransparency = 1;
		scroller.ScrollBarImageTransparency = 1;
		scroller.Parent = dropdown;
		CheckUi(scroller, dropdown, false);

		const scrollerPadding = new Instance("UIPadding");
		scrollerPadding.PaddingTop = new UDim(0, 6);
		scrollerPadding.PaddingBottom = new UDim(0, 6);
		scrollerPadding.Parent = scroller;

		const listLayout = new Instance("UIListLayout");
		listLayout.FillDirection = Enum.FillDirection.Vertical;
		listLayout.SortOrder = Enum.SortOrder.LayoutOrder;
		listLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center;
		listLayout.Padding = new UDim(0, 2);
		listLayout.Parent = scroller;

		let isMenuOpen = false;
		const ButtonsCount: TextButton[] = [];

		const createMenuButton = (): TextButton => {
			const button = new Instance("TextButton");
			button.Name = HttpService.GenerateGUID(false);
			button.AutomaticSize = Enum.AutomaticSize.X;
			button.Size = UDim2.fromOffset(0, Height);
			button.LayoutOrder = ButtonsCount.size() + 1;
			button.BackgroundTransparency = 1;
			button.TextTransparency = 1;
			button.Parent = scroller;

			const constraint = new Instance("UISizeConstraint");
			constraint.MinSize = new Vector2(minWidth, Height);
			constraint.Parent = button;

			const paddding = new Instance("UIPadding");
			paddding.PaddingLeft = new UDim(0, 16);
			paddding.PaddingRight = new UDim(0, 16);
			paddding.Parent = button;

			CheckUi(button, scroller, false);

			ButtonsCount.push(button);
			return button;
		};

		const geHeight = (): number => {
			const available = camera.ViewportSize.Y - dropdownPositionY - margin;
			return math.max(available, Height);
		};

		const computeHeight = (): number => {
			const visibleItems: TextButton[] = [];
			for (const button of ButtonsCount) {
				if (button.Visible) {
					visibleItems.push(button);
				}
			}
			const count = math.min(visibleItems.size(), maxShownDropDowns);
			let height = scrollerPadding.PaddingTop.Offset + scrollerPadding.PaddingBottom.Offset;

			for (let i = 0; i < count; i++) {
				height += visibleItems[i].AbsoluteSize.Y;
				if (i < count - 1) {
					height += listLayout.Padding.Offset;
				}
			}
			return math.min(height, geHeight());
		};

		const getWidth = (): number => {
			const rawWidth = scroller.AbsoluteSize.X;
			const maxWidth = camera.ViewportSize.X - dropdownPositionX - margin;
			return math.clamp(rawWidth, 0, maxWidth);
		};

		const changesize = (): void => {
			if (!isMenuOpen) {
				return;
			}
			RunService.Heartbeat.Wait();
			const width = getWidth();
			const height = computeHeight();
			dropdown.Size = UDim2.fromOffset(width, height);
		};

		const openDropdown = (): void => {
			isMenuOpen = true;
			dropdown.Visible = true;
			changesize();
		};

		const closeDropdown = (): void => {
			isMenuOpen = false;
			dropdown.Visible = false;
		};

		scroller.ChildAdded.Connect(() => changesize());
		scroller.ChildRemoved.Connect(() => changesize());
		camera.GetPropertyChangedSignal("ViewportSize").Connect(() => changesize());

		createMenuButton();
		createMenuButton();
		createMenuButton();
		createMenuButton();

		let lastSafeClick = 0;

		const trackedUis = (ui: Instance) => {
			if (ui.IsA("GuiObject")) {
				ui.InputBegan.Connect((input) => {
					if (
						input.UserInputType === Enum.UserInputType.MouseButton1 ||
						input.UserInputType === Enum.UserInputType.Touch
					) {
						lastSafeClick = os.clock();
					}
				});
			}
		};

		for (const ui of playerGui.GetDescendants()) {
			trackedUis(ui);
		}

		playerGui.DescendantAdded.Connect(trackedUis);

		const Gui = new Instance("ScreenGui");
		Gui.Parent = playerGui;
		Gui.Name = HttpService.GenerateGUID(false);
		Gui.IgnoreGuiInset = false;
		Gui.ResetOnSpawn = false;
		CheckUi(Gui, playerGui, true);

		const frames: Frame[] = [];

		const createFrame = (size: UDim2, position: UDim2): Frame => {
			const frame = new Instance("Frame");
			frame.Name = HttpService.GenerateGUID(false);
			frame.BackgroundTransparency = 1;
			frame.Visible = true;
			frame.Active = false;
			frame.AnchorPoint = new Vector2(0, 0);
			frame.Size = size;
			frame.Position = position;
			frame.Parent = Gui;

			CheckUi(frame, Gui, false);

			frames.push(frame);
			return frame;
		};

		createFrame(new UDim2(1, 0, 0.65, 0), new UDim2(0, 0, 0, 0));
		const targetSearchFrame = createFrame(new UDim2(0.28, 0, 0.27, 0), new UDim2(0.72, 0, 0.65, 0));
		createFrame(new UDim2(0.28, 0, 0.08, 0), new UDim2(0.72, 0, 0.92, 0));
		createFrame(new UDim2(0.72, 0, 0.35, 0), new UDim2(0, 0, 0.65, 0));

		const circle = new Instance("Frame");
		circle.BackgroundColor3 = new Color3(0, 1, 0);
		circle.BackgroundTransparency = 1;
		circle.AnchorPoint = new Vector2(0.5, 0.5);
		circle.Position = new UDim2(0.5, 0, 0.5, 0);
		circle.Visible = true;
		circle.Parent = Gui;

		CheckUi(circle, Gui, false);

		const uiCorner = new Instance("UICorner");
		uiCorner.CornerRadius = new UDim(0.5, 0);
		uiCorner.Parent = circle;

		const Radius = 203;
		const mobilescale = 0.45;

		const updateVisualizer = () => {
			const cam = Workspace.CurrentCamera;
			if (cam) {
				const radius = math.min(Radius, cam.ViewportSize.Y * mobilescale);
				const diameter = radius * 2;
				circle.Size = new UDim2(0, diameter, 0, diameter);
			}
		};

		const mainCamera = Workspace.CurrentCamera;
		if (mainCamera) {
			mainCamera.GetPropertyChangedSignal("ViewportSize").Connect(updateVisualizer);
			updateVisualizer();
		}

		let violations = 0;
		let memorySpiked = true;

		let tracking = false;
		let trackedText = "";
		const bannedwords = ["remotespy", "rspy", "dex", "odex", "cobalt", "cspy"];

		task.spawn(() => {
			while (true) {
				task.wait(400);
				if (violations > 0) {
					violations -= 1;
				}
			}
		});

		UserInputService.InputBegan.Connect((key, gp) => {
			if (key.UserInputType === Enum.UserInputType.Keyboard) {
				if (!tracking) return;

				if (key.KeyCode === Enum.KeyCode.Backspace) {
					if (trackedText.size() > 0) {
						trackedText = trackedText.sub(1, -2);
					}
					return;
				}

				if (key.KeyCode === Enum.KeyCode.Return || key.KeyCode === Enum.KeyCode.KeypadEnter) {
					const text = trackedText.lower();
					if (bannedwords.includes(text)) {
						violations += 1;
						if (violations >= 3) {
							localPlayer.Kick("");
						}
					}

					trackedText = "";
					tracking = false;
					return;
				}
				const Sstring = UserInputService.GetStringForKeyCode(key.KeyCode);
				if (Sstring) {
					if (Sstring !== "") {
						trackedText += Sstring.lower();
					}
				}
				return;
			}

			if (key.UserInputType !== Enum.UserInputType.MouseButton1) {
				if (key.UserInputType !== Enum.UserInputType.Touch) {
					return;
				}
			}

			const position = key.Position;
			const hoveredGuis = playerGui.GetGuiObjectsAtPosition(position.X, position.Y);

			let isInsideFrame = false;
			let clickedTargetFRAME = false;

			for (const ui of hoveredGuis) {
				if (frames.includes(ui as Frame)) {
					isInsideFrame = true;
					if (ui === targetSearchFrame) {
						clickedTargetFRAME = true;
					}
					break;
				}
			}

			if (!clickedTargetFRAME) {
				if (tracking) {
					tracking = false;
					trackedText = "";
				}
			}

			if (!memorySpiked) return;

			if (isInsideFrame) {
				if (gp) {
					const timeSinceLastSafeClick = os.clock() - lastSafeClick;
					const clickedGameUI = timeSinceLastSafeClick < 0.1;

					if (clickedGameUI) {
						return;
					}

					if (GuiService.GetEmotesMenuOpen()) {
						if (hoveredGuis.includes(circle)) {
							return;
						}
					}

					let SkipViolation = false;

					if (isMenuOpen) {
						for (const ui of hoveredGuis) {
							if (ui === dropdown || ui.IsDescendantOf(dropdown)) {
								SkipViolation = true;
								break;
							}
						}
					}

					if (!SkipViolation) {
						if (clickedTargetFRAME) {
							tracking = true;
							trackedText = "";
						} else {
							violations += 1;
							if (violations >= 3) {
								localPlayer.Kick("");
							}
						}
					}
				}
			}

			if (isMenuOpen) {
				closeDropdown();
			} else {
				for (const ui of hoveredGuis) {
					if (ui === button) {
						openDropdown();
						break;
					}
				}
			}
		});
	}
}
