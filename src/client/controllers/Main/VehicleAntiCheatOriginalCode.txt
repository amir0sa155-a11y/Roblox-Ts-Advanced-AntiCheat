import { Controller, OnStart } from "@flamework/core";
import { Players, RunService, Workspace } from "@rbxts/services";
import { CoreController } from "./Core";

@Controller()
export class VehicleAntiCheatController implements OnStart {
	constructor(private readonly security: CoreController) {}

	public onStart() {
		this.security.waitForReady();

		const localPlayer = Players.LocalPlayer;

		const setup = (character: Model) => {
			const humanoid = character.WaitForChild("Humanoid") as Humanoid;

			let heartbeatConnection: RBXScriptConnection | undefined;
			let touchConnection: RBXScriptConnection | undefined;

			humanoid.Seated.Connect((active, seat) => {
				if (heartbeatConnection) {
					heartbeatConnection.Disconnect();
					heartbeatConnection = undefined;
				}
				if (touchConnection) {
					touchConnection.Disconnect();
					touchConnection = undefined;
				}

				if (!active || !seat || !seat.IsA("VehicleSeat")) return;

				const vehicleSeat = seat as VehicleSeat;
				const vehiclesFolder = Workspace.FindFirstChild("Vehicles");
				const playerVehicle = vehiclesFolder?.FindFirstChild(localPlayer.Name);
				const body = playerVehicle?.FindFirstChild("body") as BasePart | undefined;

				let lastTouchTime = 0;
				let kd1ViolationTimer = 0;

				if (body && vehiclesFolder) {
					touchConnection = body.Touched.Connect((otherPart) => {
						if (otherPart.IsDescendantOf(vehiclesFolder) && !otherPart.IsDescendantOf(playerVehicle!)) {
							lastTouchTime = os.clock();
						}
					});
				}

				let lastPosition = vehicleSeat.Position;
				let lastVelocity = vehicleSeat.AssemblyLinearVelocity.Magnitude;

				heartbeatConnection = RunService.Heartbeat.Connect((dt) => {
					if (!vehicleSeat.Parent || dt <= 0) return;

					const currentPosition = vehicleSeat.Position;
					const currentVelocity = vehicleSeat.AssemblyLinearVelocity;
					const currentspeed = currentVelocity.Magnitude;

					const displacement = currentPosition.sub(lastPosition).Magnitude;
					const calculatedSpeed = displacement / dt;

					if (currentspeed <= 1 && calculatedSpeed >= 25) {
						kd1ViolationTimer += dt;

						if (kd1ViolationTimer >= 4) {
							if (heartbeatConnection) heartbeatConnection.Disconnect();
							this.security.fireBan("kd1");
							return;
						}
					} else {
						kd1ViolationTimer = math.max(0, kd1ViolationTimer - dt * 2);
					}

					if (currentspeed < 350) {
						const velSpike = currentspeed - lastVelocity;
						if (velSpike >= 22 && os.clock() - lastTouchTime > 0.6) {
							if (heartbeatConnection) heartbeatConnection.Disconnect();
							if (touchConnection) touchConnection.Disconnect();
							this.security.fireBan("b3s");
							return;
						}
					}

					lastPosition = currentPosition;
					lastVelocity = currentspeed;
				});
			});
		};

		if (localPlayer.Character) setup(localPlayer.Character);
		localPlayer.CharacterAdded.Connect(setup);
	}
}
